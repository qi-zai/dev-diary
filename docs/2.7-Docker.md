# docker 安装

---

## 安装

```sh
# 建议更新 yum
yum update

# 删除旧 docker
sudo yum remove docker\
                docker-client\
                docker-client-latest\
                docker-common\
                docker-latest\
                docker-latest-logrotate\
                docker-logrotate\
                docker-engine

# 安装 docker
yum install docker
```

## 搭建docker私有仓库

```sh
# 拉去私有仓库镜像
docker pull registry

# 运行镜像
docker run
        -d\
        -p 5000:5000\
        --restart=always\
        --name private-docker-registry\
        --privileged=true\
        -v /opt/docker-registry:/var/lib/registry\
        f32a97de94e1

# -d 后台运行
# -p 5000:5000 --> 虚拟机 : 容器
# --restart=always --> 容易随docker启动而启动，如异常终止，自动重启。docker stop会令其失效。
# --name private-docker-registry --> 设置容易名
# --privileged=true --> 使容器内真正具有root权限
# -v /opt/docker-registry:/var/lib/registry --> 映射私有仓库容器存放镜像的目录，虚拟机 : 容器
# f32a97de94e1 --> 镜像ID
```

## Jenkins
```sh
# 运行 jenkins
docker run -d \
        -p 8080:8080 \
        -p 50000:50000 \
        --name jenkins -u root \
        -v /opt/jenkins:/var/jenkins_home \
        -v /usr/share/maven:/usr/local/maven \
        -v /usr/lib/jvm/jdk1.8:/usr/local/jdk \
        cd14cecfdb3a


# 更新jenkins

# 1.下载最新jenkins包
cd /opt/jenkins
wget http://mirrors.jenkins.io/war-stable/latest/jenkins.war

# 2. 进入容器
docker exec -it -u root [ContainerId] bash

# 替换 jenkins.war 包，将以备份旧Jenkins包
whereis jenkins
cd /usr/share/jenkins/

# 3. 退出，重启
exit
docker restart [ContainerId]
```
